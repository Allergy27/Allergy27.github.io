int main() {
    std::ios::sync_with_stdio(false), std::cin.tie(nullptr);
    int _ = 1;
    // std::cin >> _;

    auto solve = [&]() -> void {
        int n;
        std::cin >> n;
        std::vector<int> qwq(n);
        for (auto &x : qwq) std::cin >> x;
        std::vector<int> ck;
        for (int i = 1; i < n; ++i) ck.emplace_back(qwq[i] > qwq[i - 1]);
        std::vector<PLL> tmp;

        tmp.push_back({ck[0], 1});
        for (int i = 1; i < n - 1; ++i) {
            if (ck[i] == tmp.back().first)
                tmp.back().second += 1;
            else
                tmp.push_back({ck[i], 1});
        }
        int len = tmp.size();
        ll ans = 0;
        int st = tmp[0].first == 0 ? 3 : 2;
        for (int i = st; i < len; i += 2) ans += tmp[i].second * tmp[i - 2].second;
        std::cout << ans << ln;
    };

    while (_--) solve();
    std::cout << std::endl;
    std::cin >> _;
    return 0;
}

int main() {
    std::ios::sync_with_stdio(false), std::cin.tie(nullptr);
    int _ = 1;
    // std::cin >> _;

    auto solve = [&]() -> void {
        const int MOD = 998244353;

        std::vector<int> fac, inv_fac;
        int max_n = 1;

        fac = {1};
        inv_fac = {1};
        max_n = 0;

        auto preprocess = [&](int n) -> void {
            if (n <= max_n) return;
            fac.resize(n + 1);
            inv_fac.resize(n + 1);
            for (int i = max_n + 1; i <= n; ++i) fac[i] = 1LL * fac[i - 1] * i % MOD;
            inv_fac[n] = 1;
            int base = fac[n], power = MOD - 2;
            while (power) {
                if (power & 1) inv_fac[n] = 1LL * inv_fac[n] * base % MOD;
                base = 1LL * base * base % MOD;
                power >>= 1;
            }
            for (int i = n - 1; i > max_n; --i) inv_fac[i] = 1LL * inv_fac[i + 1] * (i + 1) % MOD;
            max_n = n;
        };

        auto comb = [&](int n, int k) -> int {
            if (k < 0 || k > n) return 0;
            preprocess(n);  // Á°Æ‰øùÂ∑≤ÁªèÈ¢ÑÂ§ÑÁêÜÂà∞n
            return 1LL * fac[n] * inv_fac[k] % MOD * inv_fac[n - k] % MOD;
        };

        int a, b, c, d;
        std::cin >> a >> b >> c >> d;
        int ans = 0;
        for (int i = 0; i <= b; ++i) {  // Â§öÂá∫Êù•Âá†‰∏™ B ?
            int m = b - i;
            ll ab = comb(a + m - 1, a - 1);
            ll bcd = comb(c + d + i, c);
            (ans += static_cast<int>(ab * bcd % MOD)) %= MOD;
        }
        std::cout << ans << ln;
    };

    while (_--) solve();
    std::cin >> _;
    return 0;
}

int main() {
    std::ios::sync_with_stdio(false), std::cin.tie(nullptr);
    int _ = 1;
    // std::cin >> _;

    auto solve = [&]() -> void {
        int n, m;
        std::cin >> n;
        std::vector<ll> fen(n * 2 + 1);
        auto query = [&](int i) -> ll {
            ll ans = 0;
            for (int j = i; j; j -= lowbit(j)) ans += fen[j];
            return ans;
        };

        auto add = [&](int i, ll x) -> void {
            for (int j = i; j <= 2 * n; j += lowbit(j)) fen[j] += x;
        };

        auto range_sum = [&](int x, int y) -> ll { return query(y) - query(x - 1); };

        std::vector<std::vector<int>> qwq(n + 1);
        std::vector<PII> edges;
        for (int i = 0; i < n - 1; ++i) {
            int u, v;
            std::cin >> u >> v;
            edges.push_back({u, v});
            qwq[u].emplace_back(v);
            qwq[v].emplace_back(u);
        }
        std::cin >> m;
        std::vector<PII> tmp(n + 1);
        int z = 0;
        std::vector<int> depth(n + 1, -1);
        depth[1] = 0;
        auto dfs = [&](auto &&dfs, int i, int fa) -> void {
            tmp[i].first = ++z;
            for (auto x : qwq[i]) {
                if (x != fa) {
                    depth[x] = depth[i] + 1;
                    dfs(dfs, x, i);
                }
            }
            tmp[i].second = ++z;
        };
        dfs(dfs, 1, 1);
        for (int i = 1; i < 2 * n + 1; ++i) add(i, 1);
        for (int i = 0; i < m; ++i) {
            int op;
            std::cin >> op;
            if (op == 1) {
                int x, w;
                std::cin >> x >> w;
                add(tmp[x].first, w);
                add(tmp[x].second, w);
            } else {
                int y;
                std::cin >> y;
                auto [a, b] = edges[y - 1];
                y = std::max(a, b);
                std::cout << std::abs(query(2 * n) - 2 * range_sum(tmp[y].first, tmp[y].second)) / 2 << ln;
            }
        }
    };

    while (_--) solve();
    std::cout << std::endl;
    std::cin >> _;
    return 0;
}
