void solve() {
    int n, v, i, j, num, cnt = 0, a, b;
    cin >> n >> v;
    vector<int>qwq(1000005, 0);
    vector<int>qwqV(1000005, 0);
    vector<int>qwqW(1000005, 0);

            qwqW[cnt] = j * b;

             qwq[j] = max(qwq[j], qwq[j - qwqW[i]] + qwqV[i]); //Ë¶Å‰πàÂèñÔºåË¶Å‰πà‰∏çÂèñÔºå‰∏çËÄÉËôëÁ©∫Èó¥Áî®ÂÆå
    cout << qwq[v] << '\n';
