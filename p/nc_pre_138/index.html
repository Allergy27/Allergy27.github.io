    int _ = 1;
    // std::cin >> _;

    auto solve = [&]() -> void {
        int n, m;
        std::cin >> n >> m;

        std::vector<int> c(m);
        for (auto &x : r) std::cin >> x;
        for (auto &x : c) std::cin >> x;
        std::sort(r.begin(), r.end());
        std::sort(c.begin(), c.end());
        int ans = 0;
        for (int i = 0; i < m; ++i) ans += c[i] * n;

        std::vector<int> qaq(m + 1);
        for (int i = 0; i < m; ++i) qaq[i + 1] = qaq[i] + c[i];
        for (int i = 0; i < n; ++i) {
            int p = std::lower_bound(c.begin(), c.end(), r[i]) - c.begin();
            ans += r[i] * p - qaq[p];
        }
        std::cout << ans << ln;
    };

    while (_--) solve();
    std::cin >> _;
    return 0;
}

    };

    while (_--) solve();
    std::cin >> _;

        int end = 720720;  // LCM of 1-16 : 16 * 9 * 5 * 7 * 11 * 13

        std::vector<int> find(16);
        for (int i = 0; i < 16; ++i) find[i] = (1 << (i + 1)) - 1;

        for (int i = 0; i < n; ++i) {
            std::string s;
            std::cin >> s;
            int sz = s.size(), tmp = 0;
            for (int x = 0; x < sz; ++x) tmp |= (s[x] - '0') << x;
            find[sz - 1] &= tmp;
        }
        for (int i = 0; i < end; ++i) {
            bool flag = true;
            for (int x = 0; x < 16; ++x) {
                int tmp = i % (x + 1);
                if (!(find[x] & (1 << tmp))) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                std::cout << i + 1 << ln;
                return;
            }
        }
        std::cout << -1 << ln;
    };

    while (_--) solve();
    std::cin >> _;
    return 0;
}

int main() {
    std::ios::sync_with_stdio(false), std::cin.tie(nullptr);
    int _ = 1;
    std::cin >> _;

    auto sieve = [](int n) -> std::pair<std::vector<int>, std::vector<bool>> {
        std::vector<int> prime;
        std::vector<bool> not_prime(n + 1);
        for (int i = 2; i <= n; ++i) {
            if (!not_prime[i]) prime.emplace_back(i);
            for (auto x : prime) {
                if (i * x > n) break;
                not_prime[i * x] = true;
                if (i % x == 0) break;
            }
        }
        return {prime, not_prime};
    };
    auto [prime, not_prime] = sieve(270000);
    for (int i = 0; i < 3; ++i) not_prime[i] = true;
    auto solve = [&]() {
        int n, m;
        std::cin >> n;
        for (auto x : prime) {
            if (n < 8 || x >= n) {
                std::cout << -1 << ln;
                return;
            }
            if (!not_prime[n - x] && !not_prime[x]) {
                int tmp = 1;
                for (int i = 0; i < n; ++i) {
                    std::cout << tmp << ' ';
                    ((tmp += x - 1) %= n) += 1;
                }
                std::cout << ln;
                return;
            }
        }
    };

    while (_--) solve();
    std::cin >> _;
}
