void solve() {
    int n, m;
    cin >> n;
    vector<int> qwq(n);
    for (auto &x : qwq) cin >> x;
    // Step 1. O(n)é¢„å¤„ç†
    vector<int> qaq(n + 1);   // å‰ç¼€å’Œ
    vector<int> h(n + 1);     // çºµå‘å‰ç¼€å’Œ
    vector<int> w(n + 1);     // æ¨ªå‘å‰ç¼€å’Œ
    vector<int> tsum(n + 1);  // å‰ç¼€å’Œæ€»å’Œçš„å‰ç¼€å’Œ
    for (int i = 1; i <= n; ++i) qaq[i] = qaq[i - 1] + qwq[i - 1];
    for (int i = 1; i <= n; ++i) h[i] = h[i - 1] + qaq[i];
    for (int i = 1; i <= n; ++i) w[i] = w[i - 1] + qwq[i - 1] * (n - i + 1);
    auto t = [&](int l, int r) -> int {  // lambdaè¡¨è¾¾å¼ï¼Œç­‰æ•ˆä¸€ä¸ªå‡½æ•°
        return h[r] - w[l - 1] + (n - r) * qaq[l - 1];
    };
    for (int i = 1; i <= n; ++i) tsum[i] = tsum[i - 1] + t(i, n);
    // Step 2. O(log n)æŸ¥è¯¢
    cin >> m;
    vector<int> nums(n + 1);
    for (int i = 1; i <= n; ++i) nums[i] = nums[i - 1] + n - i + 1;
    while (m--) {
        int l, r;
        cin >> l >> r;
        int lf = lower_bound(nums.begin(), nums.end(), l) - nums.begin();
        int rf = lower_bound(nums.begin(), nums.end(), r) - nums.begin();
        int nl = l - nums[lf - 1] - 1;
        int nr = r - nums[rf - 1] - 1;
        if (lf == rf) {  // åœ¨åŒä¸€æ®µç›´æ¥æ±‚å’Œ
            cout << t(lf, lf + nr) - t(lf, lf + nl - 1) << ln;
        } else {
            int ans = 0;
            ans += t(lf, n) - t(lf, lf + nl - 1);  // låˆ°lå³ç«¯ç‚¹å’Œ
            ans += tsum[rf - 1] - tsum[lf];        // ä¸­é—´æ®µå’Œ
            ans += t(rf, rf + nr);                 // råˆ°rå·¦ç«¯ç‚¹å’Œ
            cout << ans << ln;
        }
    }
}

def solve():

        rf = bisect.bisect_left(nums, r)
        nl = l - nums[lf - 1] - 1
        nr = r - nums[rf - 1] - 1
        if lf == rf:  # åœ¨åŒä¸€æ®µç›´æ¥æ±‚å’Œ
            print(t(lf, lf + nr) - t(lf, lf + nl - 1))
        else:
            ans = 0
            ans += t(lf, n) - t(lf, lf + nl - 1)  # låˆ°lå³ç«¯ç‚¹å’Œ
            ans += tsum[rf - 1] - tsum[lf]        # ä¸­é—´æ®µå’Œ
            ans += t(rf, rf + nr)                 # råˆ°rå·¦ç«¯ç‚¹å’Œ
            print(ans)
