fn solve() {
    let (n, b, c) = cin!(i64, i64, i64);

    if b == 0 {
        if c >= n {
            println!("{}", n - 2);
        } else if c >= n - 2 {
            println!("{}", n - 1);
        } else {
            println!("-1");
        }
    } else if c >= n {
        println!("{}", n);
    } else {
        println!("{}", n - 0.max(1 + (n - c - 1) / b));
    }
}

fn solve() {
    let (n, m, v) = cin!(usize, usize, usize);
    let qwq = cin!([usize; n]);
    let mut qaq = vec![0; n + 1];
    for i in 0..n {
        qaq[i + 1] = qaq[i] + qwq[i];
    }
    let query = |l: usize, r: usize| qaq[r] - qaq[l];
    // ä»å‰å¾€ååˆ‡ï¼Œç¬¬mçš„å³ç«¯ç‚¹ï¼ˆå¼€
    let mut f = vec![0; m + 1];
    let mut tmp = 0;
    for i in 1..=m {
        while tmp as usize <= n && query(f[i - 1] as usize, tmp as usize) < v {
            tmp += 1;
        }
        f[i] = tmp;
    }
    // ä»åå¾€å‰åˆ‡ï¼Œç¬¬mçš„å·¦ç«¯ç‚¹ï¼ˆé—­
    let mut g = vec![n as i32; m + 1];
    tmp = n as i32;
    for i in 1..=m {
        while tmp >= 0 && query(tmp as usize, g[i - 1] as usize) < v {
            tmp -= 1;
        }
        g[i] = tmp;
    }
    let mut ans = -1;
    for i in 0..=m {

        }
    }
    println!("{ans}");
}
